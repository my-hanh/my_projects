65       def stabilize(self, prev_pts, curr_pts):
66           if prev_pts is None or curr_pts is None or len(prev_pts) < 3:
67               M = np.float32([[1, 0, 0],
68                               [0, 1, 0]])
69               raw_delta = (0, 0)
70               smooth_delta = (0, 0)
71               return M, raw_delta, smooth_delta

73           prev_pts = prev_pts.reshape(-1, 1, 2).astype(np.float32)
74           curr_pts = curr_pts.reshape(-1, 1, 2).astype(np.float32)
75   
76           T, inliers = cv2.estimateAffinePartial2D(
77               prev_pts, curr_pts,

84                               [0, 1, 0]])

88           da = np.arctan2(T[1, 0], T[0, 0])

90           raw_delta = (dx, dy)

92           self.cumulative_x += dx
93           self.cumulative_y += dy
94           self.cumulative_a += da
95   
96           curr_traj = np.array([self.cumulative_x,
97                                 self.cumulative_y,
98                                 self.cumulative_a])
99           self.trajectory.append(curr_traj)

101          radius = self.smoothing_radius
102          idx = len(self.trajectory) - 1
103  
104          start = max(0, idx - radius)
105          end = min(len(self.trajectory) - 1, idx + radius)
106  
107          window = np.array(self.trajectory[start:end + 1])
108          smooth_x, smooth_y, smooth_a = np.mean(window, axis=0)

112          diff_a = smooth_a - self.cumulative_a

118          corrected_a = da + diff_a

130      def fix_border(self, frame):
131          """ 
132           scales the image slightly (1.04x) to hide black borders created by stabilization. 
133           Reference: DataHacker Tutorial "fixBorder" 
134          """
135          s = frame.shape
136          # Scale image 4% to hide borders
137          T = cv2.getRotationMatrix2D((s[1]/2, s[0]/2), 0, 1.04)
138          frame = cv2.warpAffine(frame, T, (s[1], s[0]))
139          return frame
